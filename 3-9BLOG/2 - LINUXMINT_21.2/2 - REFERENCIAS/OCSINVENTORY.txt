PASSO A PASSO:
$ sudo tar –xvzf Ocsinventory-Unix-Agent-2.x.y.tar.gz
$ sudo cd Ocsinventory-Unix-Agent-2.x.y
$ sudo perl Makefile.PL
$ sudo make
$ sudo make install

## Após instalação editar:
$ sudo nano /var/lib/ocsinventory-agent/http:_ENDEREÇOIP_ocsinventory/ocsinv.adm
<?xml version="1.0" encoding="UTF-8"?>
<ADM>
 <ACCOUNTINFO>
  <KEYNAME>TAG</KEYNAME>

$ sudo nano /etc/ocsinventory/ocsinventory-agent.cfg
basevardir=/var/lib/ocsinventory-agent
server=http://ENDEREÇO_IP/ocsinventory
#tag=NA
#snmptimeout=3
debug=1
#snmp=1
nosoftware=0
#snmpretry=2
